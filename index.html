// WAJIB DIGANTI SETIAP ADA UPDATE KODE (contoh: 1.0.1 -> 1.0.2) const CACHE_VERSION = '1.0.1'; const CACHE_NAME = `tugas-v6-cache-${CACHE_VERSION}`; const STATIC_ASSETS = [ './', './index.html', './manifest.json', './icon-192.png', './icon-512.png' ]; // Instalasi: Paksa SW baru langsung masuk tanpa menunggu tab ditutup self.addEventListener('install', (event) => { self.skipWaiting(); event.waitUntil( caches.open(CACHE_NAME).then((cache) => cache.addAll(STATIC_ASSETS)) ); }); // Aktivasi: Eksekusi pembersihan (Clear Cache) aset versi sebelumnya self.addEventListener('activate', (event) => { event.waitUntil( caches.keys().then((keys) => Promise.all( keys.filter((key) => key.startsWith('tugas-v6-cache-') && key !== CACHE_NAME) .map((key) => caches.delete(key)) )).then(() => self.clients.claim()) // Ambil alih kontrol klien seketika ); }); // Intersepsi Jaringan: Network First untuk HTML, Cache First untuk CDN self.addEventListener('fetch', (event) => { const { request } = event; const url = new URL(request.url); // 1. Dokumen HTML & Permintaan Navigasi -> Network First if (request.mode === 'navigate' || url.pathname.endsWith('index.html')) { event.respondWith( fetch(request).then((networkResponse) => { return caches.open(CACHE_NAME).then((cache) => { cache.put(request, networkResponse.clone()); return networkResponse; }); }).catch(() => caches.match(request)) ); return; } // 2. Aset Statis & CDN -> Stale-While-Revalidate event.respondWith( caches.match(request).then((cachedResponse) => { const fetchPromise = fetch(request).then((networkResponse) => { if (networkResponse && networkResponse.status === 200) { caches.open(CACHE_NAME).then((cache) => cache.put(request, networkResponse.clone())); } return networkResponse; }).catch(() => { // Mode offline absolut }); return cachedResponse || fetchPromise; }) ); });
